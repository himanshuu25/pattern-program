//....................check lowercase or upercase.....................

 #include<iostream>
 using namespace std;
 int main(){
    char ch='5';
    if(ch>='a' && ch<='z'){
        cout<<ch<<" is a lowercase character";
    }
    else if (ch>='A' && ch<='Z'){
        cout<<ch<<" is a upercase character";
    }
    else{
        cout<<ch<<" is a invalid character";
    }
    return 0;
 }
